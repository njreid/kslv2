      type string
      format hostname
      doc:summary "Hostname value"
    }

      prop "host" required ref="hostname"

    node "service" required {

        prop "mode" required {

          prop "mode" {
            const tcp
          }
        }
        then {
          prop "port" required {
            type integer
          }
